void loop() {
  
  float current;
  
  float ar_i_min_day[24] = {0};
  float ar_i_max_day[24] = {0};
  float ar_i_rms_day[24] = {0};
  float ar_P_count_day[24] = {0};
  short ar_hour[24] = {0};
  char time_min[24] = {0};
  char time_max[24] = {0};
  //надо ещё сделать 2 массива для указания времени минимальной и максимальной нагрузок

  //переменная времени
  time_t mytime = time(NULL);
  char * time_str = ctime(&mytime);
  time_str[strlen(time_str)-1] = '\0';

  //цикл дня
  for(int i = 0; i < 24; ++i){
    float I_min = 100;
    float I_max = 0;
    double I_rms = 0;
    float P_count = 0;
    char time_i_min[10] = {0};
    char time_i_max[10] = {0};

    
    //цикл часа
    for(int j = 0; j < COUNT_MEASURE; ++j){

      //считываем значение с аналогово входа и вычисляем ток в сети
      current = analogRead(A0) / RESISTANCE / TRANS_COEFF;

      //вывод показаний на монитор ввиде графика, для контроля работы
      Serial.println(current);

      //поиск минимума по току
      if(I_min > current) {
        I_min = current;
        *(time_str+19) = '\0';
        char* ptr_hour = time_str + 11;
        strcpy(time_i_min, ptr_hour);
        
        //Отсюда берём I_min и time_i_min
      }

      //поиск максимального тока
      if(I_max < current){
        I_max = current;
        *(time_str+19) = '\0';
        char* ptr_hour = time_str + 11;
        strcpy(time_i_max, ptr_hour);
        
        //Отсюда берём I_max и time_i_max
      }
      
      //Вычисляем сумму квадратов измеряемых значений
      I_rms += pow(current, 2);

      //Суммарная потребляемая мощность
      P_count += current * VOLTAGE_MAIN;

      //добавим тревогу если значение тока превышает допустимый уровень
      if(I_max < MAX_CURRENT_VAL) {
        printf("ALARM: current is more then 19 A !!!");

      }

    }
    //Вычислим средне квадратичное показание тока за 1 час
    I_rms = pow((I_rms / COUNT_MEASURE), 0.5);

    //Занесём вычисленные параметры сети в массивы
    ar_i_min_day[i] = I_min;
    ar_i_max_day[i] = I_max;
    ar_i_rms_day[i] = I_rms;
    ar_P_count_day[i] = P_count;
    //time_min[i] = time_i_min;
